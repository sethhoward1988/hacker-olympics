public class Fibonacci {
	static double fibo(int j) {
		if (j >= 2) {
			double[] a = new double[j + 1];
			a[1] = a[2] = 1;
			for (int i = 3; i <= j; i++)
				a[i] = a[i - 1] + a[i - 2];
			return a[j];
		} else
			return j;
	}
